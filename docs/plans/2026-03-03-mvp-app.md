# Fogchaser MVP App — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a live fog prediction web app — every 6 hours, fetch real HRRR + ASOS data, run the XGBoost model, generate fog probability maps for the next 12 hours, and serve them on an interactive Leaflet map so a photographer can decide whether to get up tomorrow morning.

**Architecture:** A Python pipeline fetches live HRRR (via Herbie) and ASOS (via IEM API), runs `run_inference_hour.py` for the next 12 forecast hours, and writes GeoTIFF fog maps + weather JSON files into `app/data/`. A GitHub Actions workflow runs this pipeline every 6 hours and commits the updated files to the repo. Vercel auto-deploys on every commit, so the static frontend always shows the latest forecast.

**Tech Stack:** Python (Herbie, xgboost, rasterio, requests), Leaflet.js + georaster-layer-for-leaflet + geoblaze (CDN, no build step), GitHub Actions (free), Vercel (free static hosting)

```python
def infer_from_dataframes(asos_feat: pd.DataFrame, hrrr_feat: pd.DataFrame,
                           valid_dt: datetime):
    """
    Core inference: merge ASOS + HRRR features, run XGBoost + Platt calibration.

    Parameters
    ----------
    asos_feat : DataFrame with columns:
        station, elevation, t_td_spread_lag, wind_speed_mph_lag,
        drct_sin_lag, drct_cos_lag, hour_sin, hour_cos, month_sin, month_cos
    hrrr_feat : DataFrame with columns:
        station, hpbl_m, tcdc_bl_pct, dpt2m_k, tmp2m_k, rh2m_pct,
        hgt_cldbase_m, u10_ms, v10_ms
    valid_dt  : prediction datetime (UTC)

    Returns
    -------
    cal_probs : np.ndarray of calibrated probabilities, one per station
    stations  : list of station codes in same order
    """
    merged = pd.merge(asos_feat, hrrr_feat, on="station", how="inner")

    with open(STATION_LIST) as f:
        station_list = json.load(f)
    merged = encode_station(merged, station_list)

    with open(CALIBRATOR_PATH, "rb") as f:
        cal_dict = pickle.load(f)

    feature_cols = cal_dict["features"]
    fill_medians = cal_dict["fill_medians"]

    for col in feature_cols:
        if col not in merged.columns:
            merged[col] = fill_medians[col]

    X = merged[feature_cols].fillna(fill_medians)

    model = xgb.XGBClassifier()
    model.load_model(MODEL_PATH)
    raw_scores = model.predict_proba(X)[:, 1]

    calibrator = cal_dict["calibrator"]
    cal_probs = calibrator.predict_proba(raw_scores.reshape(-1, 1))[:, 1]

    return cal_probs, merged["station"].tolist()
```

```python
"""
Fetch the most recent ASOS observations for DC metro stations from IEM.

Returns a DataFrame of lag features ready for inference.
"""
import numpy as np
import pandas as pd
import requests
from datetime import datetime, timezone, timedelta
from io import StringIO

DC_STATIONS = ["KBWI", "KCGS", "KDCA", "KFDK", "KGAI", "KHEF", "KIAD", "KNYG"]

IEM_URL = "https://mesonet.agron.iastate.edu/cgi-bin/request/asos.py"


def fetch_asos_latest(lookback_hours: int = 2,
                      stations: list = DC_STATIONS) -> pd.DataFrame:
    """
    Fetch recent ASOS observations and return lag feature DataFrame.

    Parameters
    ----------
    lookback_hours : how many hours back to search for the most recent obs
    stations       : list of ICAO station codes

    Returns
    -------
    DataFrame with columns: station, elevation, t_td_spread_lag,
        wind_speed_mph_lag, drct_sin_lag, drct_cos_lag
    Returns empty DataFrame if the request fails — caller must handle this.
    """
    now_utc = datetime.now(timezone.utc)
    start = now_utc - timedelta(hours=lookback_hours)

    params = {
        "data": "tmpf,dwpf,sknt,drct,vsby",
        "tz": "UTC",
        "format": "comma",
        "latlon": "yes",
        "elev": "yes",
        "report_type": "3",          # routine METAR reports only
        "year1": start.year, "month1": start.month, "day1": start.day,
        "hour1": start.hour, "minute1": 0,
        "year2": now_utc.year, "month2": now_utc.month, "day2": now_utc.day,
        "hour2": now_utc.hour, "minute2": 59,
    }
    for s in stations:
        params.setdefault("station", [])
        if isinstance(params["station"], list):
            params["station"].append(s)
        else:
            params["station"] = [params["station"], s]

    try:
        resp = requests.get(IEM_URL, params=params, timeout=30)
        resp.raise_for_status()
    except Exception as e:
        print(f"  WARNING: ASOS fetch failed: {e}")
        return pd.DataFrame()

    # IEM prepends comment lines starting with "#" — skip them
    lines = [l for l in resp.text.splitlines() if not l.startswith("#")]
    if len(lines) < 2:
        print("  WARNING: ASOS response empty")
        return pd.DataFrame()

    df = pd.read_csv(StringIO("\n".join(lines)))
    df.columns = df.columns.str.strip()
    df["valid"] = pd.to_datetime(df["valid"], utc=True, errors="coerce")
    df = df.dropna(subset=["valid"])

    # Keep most recent observation per station
    df = df.sort_values("valid").groupby("station").last().reset_index()

    # Compute lag features (same names as training data)
    df["t_td_spread_lag"] = (df["tmpf"] - df["dwpf"]).clip(lower=0)
    df["wind_speed_mph_lag"] = pd.to_numeric(df["sknt"], errors="coerce") * 1.15078
    drct_rad = np.radians(pd.to_numeric(df["drct"], errors="coerce").fillna(0))
    df["drct_sin_lag"] = np.sin(drct_rad)
    df["drct_cos_lag"] = np.cos(drct_rad)
    df["elevation"] = pd.to_numeric(df["elev"], errors="coerce")

    return df[["station", "elevation", "t_td_spread_lag",
               "wind_speed_mph_lag", "drct_sin_lag", "drct_cos_lag"]].copy()
```

```python
"""
Fetch HRRR forecast variables for DC metro stations for one forecast hour.

Uses Herbie to download only the needed variables via byte-range requests.
"""
import numpy as np
import pandas as pd
import xarray as xr
from datetime import datetime, timezone
from herbie import Herbie

# Station coordinates (lat, lon in WGS84)
DC_STATION_COORDS = {
    "KBWI": (39.1733, -76.6841),
    "KCGS": (38.9806, -76.9223),
    "KDCA": (38.8472, -77.0345),
    "KFDK": (39.4176, -77.3743),
    "KGAI": (39.1683, -77.1660),
    "KHEF": (38.7214, -77.5154),
    "KIAD": (38.9348, -77.4473),
    "KNYG": (38.5017, -77.3053),
}

# Output column names (must match hrrr_test_dc_2024.parquet)
REQUIRED_COLS = ["station", "hpbl_m", "tcdc_bl_pct", "dpt2m_k", "tmp2m_k",
                 "rh2m_pct", "hgt_cldbase_m", "u10_ms", "v10_ms"]

# Herbie variable search string (matches variables used in training)
HRRR_SEARCH = (
    ":(TMP|DPT|RH):2 m above ground|"
    ":UGRD:10 m above ground|"
    ":VGRD:10 m above ground|"
    ":HPBL:|"
    ":TCDC:boundary layer cloud layer|"
    ":HGT:cloud base"
)

# Internal short names Herbie assigns → our column names
RENAME_MAP = {
    "t2m": "tmp2m_k",
    "d2m": "dpt2m_k",
    "r2":  "rh2m_pct",
    "u10": "u10_ms",
    "v10": "v10_ms",
    "blh": "hpbl_m",
    "tcc": "tcdc_bl_pct",
    "gh":  "hgt_cldbase_m",
}


def extract_station_values(ds: xr.Dataset,
                            station_coords: dict) -> pd.DataFrame:
    """
    Sample an xarray Dataset at each station's nearest grid point.

    Parameters
    ----------
    ds             : merged HRRR xarray Dataset (all variables on same grid)
    station_coords : dict of {station_code: (lat, lon)}

    Returns
    -------
    DataFrame with REQUIRED_COLS, one row per station
    """
    rows = []
    for station, (lat, lon) in station_coords.items():
        # HRRR uses 0-360 longitude — convert negative lons
        lon360 = lon + 360 if lon < 0 else lon

        # Find nearest grid point using lat/lon arrays from the dataset
        grid_lats = ds["latitude"].values
        grid_lons = ds["longitude"].values  # already 0-360 in HRRR

        dist = np.sqrt((grid_lats - lat)**2 + (grid_lons - lon360)**2)
        idx = int(np.argmin(dist))

        row = {"station": station}
        for short_name, col_name in RENAME_MAP.items():
            if short_name in ds:
                row[col_name] = float(ds[short_name].values.ravel()[idx])
            else:
                row[col_name] = np.nan

        rows.append(row)

    df = pd.DataFrame(rows)

    # Fill cloud base NaN with 10,000m sentinel (NaN = no cloud present)
    df["hgt_cldbase_m"] = df["hgt_cldbase_m"].fillna(10000.0)

    return df[REQUIRED_COLS].copy()


def fetch_hrrr_forecast_hour(run_time: datetime, fxx: int,
                              station_coords: dict = DC_STATION_COORDS,
                              save_dir: str = "/tmp/hrrr_live") -> pd.DataFrame:
    """
    Fetch one HRRR forecast hour and extract station values.

    Parameters
    ----------
    run_time : HRRR model run time (UTC), e.g. datetime(2024, 12, 10, 6)
    fxx      : forecast hour offset (0 = analysis, 1 = +1h, ..., 17 = +17h)
    save_dir : local directory for Herbie to cache downloaded files

    Returns
    -------
    DataFrame with REQUIRED_COLS, one row per station.
    Returns empty DataFrame if download fails.
    """
    try:
        H = Herbie(run_time, model="hrrr", product="sfc", fxx=fxx,
                   save_dir=save_dir)
        result = H.xarray(HRRR_SEARCH)

        # Herbie returns a list when regex matches multiple variables — merge them
        if isinstance(result, list):
            ds = xr.merge(result, compat="override", join="override")
        else:
            ds = result

        # Ensure timezone-aware valid_time
        vt = ds.get("valid_time", None)
        if vt is not None and hasattr(vt, "dt"):
            if vt.dt.tz is None:
                ds["valid_time"] = vt.dt.tz_localize("UTC")

        return extract_station_values(ds, station_coords)

    except Exception as e:
        print(f"  WARNING: HRRR fetch failed (fxx={fxx}): {e}")
        return pd.DataFrame()
```

```python
"""
Generate plain-English weather summary JSON for the click panel.
"""
import json
import numpy as np
import pandas as pd
from pathlib import Path
from datetime import datetime


def fog_bullets(t_td_spread_f: float, wind_mph: float,
                rh_pct: float, hpbl_m: float) -> list:
    """Return 3-4 plain-English bullets explaining fog likelihood."""
    bullets = []

    # Dew point spread
    if t_td_spread_f <= 1.0:
        bullets.append(f"Dew point spread {t_td_spread_f:.1f}°F — air is nearly saturated")
    elif t_td_spread_f <= 2.0:
        bullets.append(f"Dew point spread {t_td_spread_f:.1f}°F — fog conditions possible")
    elif t_td_spread_f <= 4.0:
        bullets.append(f"Dew point spread {t_td_spread_f:.1f}°F — drier than typical fog conditions")
    else:
        bullets.append(f"Dew point spread {t_td_spread_f:.1f}°F — too dry for fog, unlikely")

    # Wind
    if wind_mph < 2:
        bullets.append(f"Wind {wind_mph:.0f} mph — nearly calm, ideal for fog")
    elif wind_mph <= 4:
        bullets.append(f"Wind {wind_mph:.0f} mph — light, within fog range")
    elif wind_mph <= 7:
        bullets.append(f"Wind {wind_mph:.0f} mph — breezy, may partially suppress fog")
    else:
        bullets.append(f"Wind {wind_mph:.0f} mph — too windy, mixing suppresses fog")

    # Humidity
    if rh_pct >= 95:
        bullets.append(f"Humidity {rh_pct:.0f}% — near saturation")
    elif rh_pct >= 85:
        bullets.append(f"Humidity {rh_pct:.0f}% — moderately humid")
    else:
        bullets.append(f"Humidity {rh_pct:.0f}% — relatively dry")

    # Boundary layer
    if hpbl_m < 100:
        bullets.append(f"Atmosphere very stable ({hpbl_m:.0f}m boundary layer) — traps moisture near surface")
    elif hpbl_m < 400:
        bullets.append(f"Shallow boundary layer ({hpbl_m:.0f}m) — supports fog formation")

    return bullets


def area_label(avg_prob: float) -> str:
    """Convert average grid probability to plain-language label."""
    if avg_prob < 0.10:
        return "Low"
    elif avg_prob < 0.20:
        return "Moderate"
    elif avg_prob < 0.30:
        return "High"
    else:
        return "Very High"


def build_weather_json(asos_df: pd.DataFrame, hrrr_df: pd.DataFrame,
                       cal_probs: np.ndarray, stations: list,
                       valid_dt: datetime, avg_grid_prob: float) -> dict:
    """
    Build the weather JSON dict for one forecast hour.

    Parameters
    ----------
    asos_df      : ASOS lag features DataFrame
    hrrr_df      : HRRR features DataFrame
    cal_probs    : calibrated fog probabilities per station
    stations     : station codes in same order as cal_probs
    valid_dt     : forecast valid time (UTC)
    avg_grid_prob: mean fog probability across the spatial grid

    Returns
    -------
    dict ready for json.dumps()
    """
    merged = pd.merge(asos_df, hrrr_df, on="station", how="inner")

    station_data = {}
    for sta, prob in zip(stations, cal_probs):
        row = merged[merged["station"] == sta]
        if row.empty:
            continue
        row = row.iloc[0]

        t_td = float(row.get("t_td_spread_lag", 4.0))
        wind = float(row.get("wind_speed_mph_lag", 5.0))
        rh = float(row.get("rh2m_pct", 80.0))
        hpbl = float(row.get("hpbl_m", 500.0))

        # Convert K to °F for T-Td if needed (HRRR temps are in Kelvin)
        # t_td_spread_lag is already in °F from ASOS
        station_data[sta] = {
            "fog_prob": round(float(prob), 3),
            "t_td_spread_f": round(t_td, 1),
            "wind_mph": round(wind, 1),
            "rh_pct": round(rh, 1),
            "hpbl_m": round(hpbl, 0),
            "bullets": fog_bullets(t_td, wind, rh, hpbl),
        }

    return {
        "valid_utc": valid_dt.strftime("%Y-%m-%dT%H:00:00Z"),
        "area_summary": {
            "avg_prob": round(avg_grid_prob, 3),
            "label": area_label(avg_grid_prob),
        },
        "stations": station_data,
    }


def save_weather_json(data: dict, output_dir: str, valid_dt: datetime):
    """Write weather JSON to app/data/weather_{HH}UTC.json."""
    Path(output_dir).mkdir(parents=True, exist_ok=True)
    fname = f"weather_{valid_dt.strftime('%H')}UTC.json"
    out_path = Path(output_dir) / fname
    with open(out_path, "w") as f:
        json.dump(data, f, indent=2)
    return str(out_path)
```

```python
"""
Live forecast orchestrator.

Usage: python3 scripts/run_live_forecast.py

Finds the most recent HRRR run, fetches live ASOS data, runs inference
for the next 12 forecast hours, and writes all outputs to app/data/.
"""
import json
import numpy as np
import rasterio
import sys
from datetime import datetime, timezone, timedelta
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent.parent))

from scripts.fetch_live_asos import fetch_asos_latest
from scripts.fetch_live_hrrr import fetch_hrrr_forecast_hour, DC_STATION_COORDS
from scripts.run_inference_hour import infer_from_dataframes, encode_station, CALIBRATOR_PATH
from scripts.spatial_pipeline import run_spatial_pipeline
from scripts.generate_weather_json import build_weather_json, save_weather_json

import pickle, pandas as pd

OUTPUT_DIR = "app/data"
FORECAST_HOURS = 12   # how many hours ahead to forecast
# HRRR runs at 00, 06, 12, 18 UTC — use the most recent completed run
# We allow up to 90 min for the run to post before falling back
HRRR_RUN_OFFSET_HOURS = [0, 6, 12, 18]


def latest_hrrr_run(now_utc: datetime) -> datetime:
    """Find the most recent HRRR run that should be available."""
    # HRRR takes ~45 min to post; subtract 90 min for safety
    cutoff = now_utc - timedelta(minutes=90)
    run_hour = max(h for h in HRRR_RUN_OFFSET_HOURS if h <= cutoff.hour)
    return cutoff.replace(hour=run_hour, minute=0, second=0, microsecond=0)


def read_geotiff_mean(tif_path: str) -> float:
    """Return mean fog probability from a GeoTIFF (ignores nodata cells)."""
    with rasterio.open(tif_path) as src:
        arr = src.read(1).astype(float)
        nodata = src.nodata or -9999.0
    valid = arr[(arr != nodata) & (arr > -9990)]
    return float(np.mean(valid)) if len(valid) > 0 else 0.0


def main():
    now_utc = datetime.now(timezone.utc)
    run_time = latest_hrrr_run(now_utc)
    print(f"\n{'='*60}")
    print(f"Fogchaser live forecast — {now_utc.strftime('%Y-%m-%d %H:%M UTC')}")
    print(f"HRRR run: {run_time.strftime('%Y-%m-%d %H UTC')}")
    print(f"{'='*60}\n")

    # Fetch ASOS once — use as lag features for all forecast hours
    print("Fetching live ASOS observations...")
    asos_df = fetch_asos_latest(lookback_hours=2)
    if asos_df.empty:
        print("ERROR: No ASOS data — cannot generate forecast.")
        sys.exit(1)
    print(f"  Got {len(asos_df)} stations: {sorted(asos_df['station'].tolist())}")

    with open(CALIBRATOR_PATH, "rb") as f:
        cal_dict = pickle.load(f)

    manifest = {
        "generated_utc": now_utc.strftime("%Y-%m-%dT%H:%M:%SZ"),
        "hrrr_run_utc": run_time.strftime("%Y-%m-%dT%H:00:00Z"),
        "hours": [],
    }

    Path(OUTPUT_DIR).mkdir(parents=True, exist_ok=True)

    for fxx in range(FORECAST_HOURS):
        valid_dt = run_time + timedelta(hours=fxx)
        hour_str = valid_dt.strftime("%H UTC")
        print(f"\n[fxx={fxx:02d}] Valid: {valid_dt.strftime('%Y-%m-%d %H UTC')}")

        # Add hour/month cyclic features to ASOS (based on this forecast hour)
        asos_hour = asos_df.copy()
        asos_hour["hour_sin"]  = np.sin(2 * np.pi * valid_dt.hour / 24)
        asos_hour["hour_cos"]  = np.cos(2 * np.pi * valid_dt.hour / 24)
        asos_hour["month_sin"] = np.sin(2 * np.pi * (valid_dt.month - 1) / 12)
        asos_hour["month_cos"] = np.cos(2 * np.pi * (valid_dt.month - 1) / 12)

        # Fetch HRRR for this forecast hour
        hrrr_df = fetch_hrrr_forecast_hour(run_time, fxx)
        if hrrr_df.empty:
            print(f"  SKIP — HRRR not available for fxx={fxx}")
            continue

        # Run inference
        try:
            cal_probs, stations = infer_from_dataframes(asos_hour, hrrr_df, valid_dt)
        except Exception as e:
            print(f"  SKIP — inference failed: {e}")
            continue

        # Run spatial pipeline
        lats = [DC_STATION_COORDS[s][0] for s in stations if s in DC_STATION_COORDS]
        lons = [DC_STATION_COORDS[s][1] for s in stations if s in DC_STATION_COORDS]
        probs_filtered = [p for s, p in zip(stations, cal_probs) if s in DC_STATION_COORDS]

        tif_path = run_spatial_pipeline(
            station_probs=probs_filtered,
            station_lats=lats,
            station_lons=lons,
            terrain_offset_path="data/processed/terrain_offset_grid.tif",
            output_dir=OUTPUT_DIR,
            valid_dt=valid_dt,
        )

        # Compute area mean for summary label
        avg_prob = read_geotiff_mean(tif_path)

        # Build + save weather JSON
        weather = build_weather_json(asos_hour, hrrr_df, cal_probs, stations,
                                      valid_dt, avg_prob)
        json_path = save_weather_json(weather, OUTPUT_DIR, valid_dt)

        tif_fname = Path(tif_path).name
        json_fname = Path(json_path).name

        manifest["hours"].append({
            "valid_utc": valid_dt.strftime("%Y-%m-%dT%H:00:00Z"),
            "label": weather["area_summary"]["label"],
            "avg_prob": round(avg_prob, 3),
            "tif": tif_fname,
            "weather_json": json_fname,
        })
        print(f"  → {tif_fname}  area_avg={avg_prob:.3f}  [{weather['area_summary']['label']}]")

    # Write manifest
    manifest_path = Path(OUTPUT_DIR) / "manifest.json"
    with open(manifest_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"\nManifest written: {manifest_path}")
    print(f"Done. {len(manifest['hours'])} forecast hours generated.")


if __name__ == "__main__":
    main()
```

```javascript
  }

  // Build slider
  slider.max = manifest.hours.length - 1;
  slider.value = 0;
  buildHourLabels();

  // Pre-fetch all weather JSON
  await Promise.all(manifest.hours.map(async (h) => {
    try {
      const r = await fetch(`${DATA_DIR}/${h.weather_json}`);
      weatherByHour[h.tif] = await r.json();
    } catch (_) {}
  }));

  // Show first hour
  await showHour(0);
  loadingEl.style.display = "none";

  // Update summary badge
  const avgProb = manifest.hours.reduce((s, h) => s + h.avg_prob, 0) / manifest.hours.length;
  summaryBadge.textContent = `Fog tonight: ${probLabel(avgProb)}`;

  slider.addEventListener("input", () => showHour(parseInt(slider.value)));

  // Click to query
  map.on("click", onMapClick);
}

function buildHourLabels() {
  hourLabels.innerHTML = "";
  manifest.hours.forEach((h, i) => {
    const span = document.createElement("span");
    span.textContent = h.valid_utc.slice(11, 13) + "Z";
    hourLabels.appendChild(span);
  });
}

// ── Show one forecast hour ────────────────────────────────────────────────────
async function showHour(idx) {
  const h = manifest.hours[idx];
  hourLabel.textContent = `${h.valid_utc.slice(0, 10)} ${h.valid_utc.slice(11, 13)}:00 UTC  —  ${h.label}`;

  // Load georaster (cached)
  const gr = await loadGeoRaster(h.tif);

  // Remove previous layer
  if (currentLayer) map.removeLayer(currentLayer);

  currentLayer = new GeoRasterLayer({
    georaster: gr,
    opacity: 0.8,
    pixelValuesToColorFn: (values) => {
      const v = values[0];
      if (v == null || v < -9990) return null;   // nodata / low-confidence
      const [r, g, b, a] = fogColor(v);
      return `rgba(${r},${g},${b},${a / 255})`;
    },
    resolution: 256,
  });
  currentLayer.addTo(map);
}

async function loadGeoRaster(filename) {
  if (georasters[filename]) return georasters[filename];
  const resp = await fetch(`${DATA_DIR}/${filename}`);
  const buffer = await resp.arrayBuffer();
  const gr = await parseGeoraster(buffer);
  georasters[filename] = gr;
  return gr;
}

// ── Click handler ─────────────────────────────────────────────────────────────
async function onMapClick(e) {
  const { lat, lng } = e.latlng;
  const h = manifest.hours[parseInt(slider.value)];
  const gr = await loadGeoRaster(h.tif);

  // Query raster value at click point
  let prob = null;
  try {
    const results = await geoblaze.identify(gr, [lng, lat]);
    if (results && results[0] != null && results[0] > -9990) {
      prob = results[0];
    }
  } catch (_) {}

  if (prob == null) return;

  // Find nearest station for weather bullets
  const weather = weatherByHour[h.tif];
  const bullets = getNearestBullets(lat, lng, weather);

  // Populate panel
  document.getElementById("prob-value").textContent = `${Math.round(prob * 100)}%`;
  document.getElementById("prob-label").textContent = probLabel(prob);
  const ul = document.getElementById("bullets");
  ul.innerHTML = "";
  bullets.forEach(b => {
    const li = document.createElement("li");
    li.textContent = b;
    ul.appendChild(li);
  });

  clickPanel.classList.add("visible");
}

// Station coordinates (must match scripts/fetch_live_hrrr.py)
const STATION_COORDS = {
  KBWI: [39.1733, -76.6841], KCGS: [38.9806, -76.9223],
  KDCA: [38.8472, -77.0345], KFDK: [39.4176, -77.3743],
  KGAI: [39.1683, -77.1660], KHEF: [38.7214, -77.5154],
  KIAD: [38.9348, -77.4473], KNYG: [38.5017, -77.3053],
};

function getNearestBullets(lat, lng, weather) {
  if (!weather || !weather.stations) return ["Weather data unavailable"];
  let nearest = null, minDist = Infinity;
  for (const [sta, coords] of Object.entries(STATION_COORDS)) {
    const d = Math.hypot(coords[0] - lat, coords[1] - lng);
    if (d < minDist && weather.stations[sta]) { minDist = d; nearest = sta; }
  }
  return nearest ? weather.stations[nearest].bullets : ["Weather data unavailable"];
}
```
